        <h1>FPB dan KPK</h1>
        <h4>Faktor Persekutuan Besar & Kelipatan Persekutuan Kecil</h4>
        <p>Masukkan angka 1:
            <input type="number" id="angka1"/>
        </p>
        <p>Masukkan angka 2:
            <input type="number" id="angka2"/>
        </p>
        <button onclick="cal()">Hitung FPB & KPK</button>
        <h2>
            FPB = <span id="fpb"> </span><br>
            KPK = <span id="kpk"> </span>
        </h2>
        <script>
            function cal(){
                var hitung={
                    num1: parseInt(document.getElementById('angka1').value),
                    num2: parseInt(document.getElementById('angka2').value)
                }
                var FPB=0;
                for(var c=1; c<=hitung.num1; c++){
                    var d=hitung.num1%c
                    var e=hitung.num2%c
                    if(d==e && e==0){
                        var FPB=c
                    }
                }
                var KPK=(hitung.num1*hitung.num2)/FPB
                document.getElementById('fpb').innerHTML=FPB
                document.getElementById('kpk').innerHTML=KPK
            }
        </script>
